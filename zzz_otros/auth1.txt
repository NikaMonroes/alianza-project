<?php
if (session_status() === PHP_SESSION_NONE) {
    session_start();
}
require_once '../db.php';

// Funci칩n para registrar un usuario con hash de contrase침a
function registrarUsuario($nombre, $email, $password, $rol, $conn) {
    $hashedPassword = password_hash($password, PASSWORD_DEFAULT);
    $stmt = $conn->prepare("INSERT INTO usuarios (nombre, email, password, rol) VALUES (?, ?, ?, ?)");
    $stmt->bind_param("ssss", $nombre, $email, $hashedPassword, $rol);
    return $stmt->execute();
}

// Funci칩n para autenticar usuario
function autenticarUsuario($email, $password, $conn) {
    $stmt = $conn->prepare("SELECT id, nombre, password, rol FROM usuarios WHERE email = ?");
    $stmt->bind_param("s", $email);
    $stmt->execute();
    $result = $stmt->get_result();
    
    if ($result->num_rows === 1) {
        $usuario = $result->fetch_assoc();
        if (password_verify($password, $usuario['password'])) {
            $_SESSION['usuario_id'] = $usuario['id'];
            $_SESSION['usuario_nombre'] = $usuario['nombre'];
            $_SESSION['usuario_rol'] = $usuario['rol'];
            return true;
        }
    }
    return false;
}

// Cerrar sesi칩n
function cerrarSesion() {
    session_unset();
    session_destroy();
    header("Location: ../usuarios/login.php");
    exit();
}

?>
