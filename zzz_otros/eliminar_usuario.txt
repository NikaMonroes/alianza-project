<?php
require_once '../db.php';

if (isset($_GET['id'])) {
    $id = $_GET['id'];

    try {
        $stmt = $pdo->prepare("DELETE FROM usuarios WHERE id = ?");
        $stmt->execute([$id]);

        header("Location: registrar.php?mensaje=Usuario eliminado correctamente");
        exit;
    } catch (PDOException $e) {
        header("Location: registrar.php?error=" . urlencode($e->getMessage()));
        exit;
    }
}
?>
