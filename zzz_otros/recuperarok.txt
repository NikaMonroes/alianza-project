<?php
session_start();
require_once '../includes/db.php';
require_once '../includes/emailservice.php';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $email = $_POST['email'];

    // Buscar usuario
    $stmt = $pdo->prepare("SELECT id FROM usuarios WHERE email = ?");
    $stmt->execute([$email]);
    $usuario = $stmt->fetch();

    if ($usuario) {
        // Generar token aleatorio
        $token = bin2hex(random_bytes(32));

        // Guardar token en la BD
        $expiracion = date("Y-m-d H:i:s", strtotime("+1 hour")); // Token expira en una hora

        $stmt = $pdo->prepare("UPDATE usuarios SET token_recuperacion = ?, token_expiracion = ? WHERE id = ?");
        $stmt->execute([$token, $expiracion, $usuario['id']]);

        // Enviar email con el enlace de recuperación
        $url = "http://localhost/cafeteria/usuarios/reset.php?token=$token";
        $asunto = "Recuperación de contraseña";
        $mensaje = "<p>Hiciste una solicitud de recuperación de contraseña. Haz clic en el siguiente enlace:</p>
                    <a href='$url'>Restablecer contraseña</a>";

        enviarCorreo($email, $asunto, $mensaje);

        echo "Correo de recuperación enviado.";
    } else {
        echo "El correo no está registrado.";
    }
}

                    <input type="email" name="email" id="email" class="form-control" placeholder="tuemail@example.com" required>
