<?php
require __DIR__ . '/../vendor/autoload.php';

use Dompdf\Dompdf;
use Dompdf\Options;

session_start();

if (!isset($_SESSION['ultima_venta'])) {
    echo "No hay datos para generar la factura.";
    exit;
}

$venta = $_SESSION['ultima_venta'];
$productos = $venta['productos'];
$total = $venta['total'];
$fecha = $venta['fecha'];

// Datos adicionales (puedes ajustar)
$vendedor = $_SESSION['usuario'] ?? 'Vendedor Genérico';
$factura_numero = rand(1000, 9999); // o puedes llevarlo en BD
$pago_cliente = $_SESSION['pago_cliente'] ?? $total;
$devolucion = $pago_cliente - $total;

// Estilos + contenido HTML
$html = '
<style>
    body { font-family: Helvetica, sans-serif; font-size: 12px; }
    h1, h3, p { text-align: center; margin: 5px 0; }
    .info { margin-bottom: 20px; }
    table { width: 100%; border-collapse: collapse; margin-bottom: 20px; }
    th, td { border: 1px solid #999; padding: 8px; text-align: center; }
    th { background-color: #f2f2f2; }
    .total { font-weight: bold; }
</style>

<h1>Cafetería Alianza</h1>
<h3>Factura de Venta</h3>
<p>Factura N.º <strong>' . $factura_numero . '</strong></p>
<p>Fecha: ' . $fecha . '</p>
<p>Vendedor: ' . htmlspecialchars($vendedor) . '</p>

<table>
    <thead>
        <tr>
            <th>Producto</th>
            <th>Cantidad</th>
            <th>Precio Unitario</th>
            <th>Subtotal</th>
        </tr>
    </thead>
    <tbody>';

foreach ($productos as $item) {
    $subtotal = $item['precio'] * $item['cantidad'];
    $html .= "<tr>
        <td>" . htmlspecialchars($item['nombre']) . "</td>
        <td>{$item['cantidad']}</td>
        <td>$" . number_format($item['precio'], 0, ',', '.') . "</td>
        <td>$" . number_format($subtotal, 0, ',', '.') . "</td>
    </tr>";
}

$html .= '
        <tr>
            <td colspan="3" class="total">Total a pagar</td>
            <td class="total">$' . number_format($total, 0, ',', '.') . '</td>
        </tr>
        <tr>
            <td colspan="3">Pago del cliente</td>
            <td>$' . number_format($pago_cliente, 0, ',', '.') . '</td>
        </tr>
        <tr>
            <td colspan="3">Devolución</td>
            <td>$' . number_format($devolucion, 0, ',', '.') . '</td>
        </tr>
    </tbody>
</table>

<p>¡Gracias por tu compra!</p>
<p>Esperamos verte pronto!</p>
';

// Configurar Dompdf
$options = new Options();
$options->set('defaultFont', 'Helvetica');
$dompdf = new Dompdf($options);
$dompdf->loadHtml($html);
$dompdf->setPaper('A4', 'portrait');
$dompdf->render();

// Descargar
$dompdf->stream("factura_" . date('Ymd_His') . ".pdf", ["Attachment" => false]); // cambiar a true para descargar directamente, con false lo visualiza en el navegador

unset($_SESSION['ultima_venta'], $_SESSION['pago_cliente']);
?>
